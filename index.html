While Loop

Executes code repeatedly while a condition is true

let i = 0;
while (i < 5) {
    console.log("Count: " + i);
    i++; // Don't forget to increment!
}
Output:
Count: 0
Count: 1
Count: 2
Count: 3
Count: 4

Do-While Loop

Executes code at least once, then repeats while condition is true

let i = 0;
do {
    console.log("This runs at least once: " + i);
    i++;
} while (i < 3);
Output:
This runs at least once: 0
This runs at least once: 1
This runs at least once: 2
Key Difference: Do-while runs the code first, then checks the condition!

For Loop

Perfect for when you know exactly how many times to loop

for (let i = 0; i < 5; i++) {
    console.log("Iteration: " + i);
}
Initialize
Condition
Increment
Output:
Iteration: 0
Iteration: 1
Iteration: 2
Iteration: 3
Iteration: 4

For...of Loop

Perfect for iterating through arrays and strings

const fruits = ["apple", "banana", "orange"];

for (const fruit of fruits) {
    console.log("I love " + fruit);
}
Output:
I love apple
I love banana
I love orange
Bonus: Works with strings too!

JavaScript Strings

Strings are sequences of characters used to represent text

let singleQuotes = 'Hello World';
let doubleQuotes = "Hello World";
let backticks = `Hello World`;

// String concatenation
let firstName = "John";
let lastName = "Doe";
let fullName = firstName + " " + lastName;
Result: "John Doe"

Template Literals 🎯

Use backticks (`) for powerful string formatting with ${} syntax

const name = "Alice";
const age = 25;
const score = 95;

const message = `Hello ${name}!
You are ${age} years old.
Your score is ${score}%.
Grade: ${score >= 90 ? 'A' : 'B'}`;

console.log(message);
Output:
Hello Alice!
You are 25 years old.
Your score is 95%.
Grade: A

Useful String Methods

const text = "JavaScript is Amazing";

console.log(text.length);        // 21
console.log(text.toUpperCase()); // "JAVASCRIPT IS AMAZING"
console.log(text.toLowerCase()); // "javascript is amazing"
console.log(text.charAt(0));       // "J"
console.log(text.indexOf("Script")); // 4
console.log(text.slice(0, 10));    // "JavaScript"
Pro Tip: String methods don't modify the original string - they return a new one!

Loops + Strings = Power! 💪

const word = "HELLO";

// Using for...of to iterate through each character
for (const letter of word) {
    console.log(`Letter: ${letter}`);
}

// Using regular for loop
for (let i = 0; i < word.length; i++) {
    console.log(`Position ${i}: ${word[i]}`);
}
Output:
Letter: H, Letter: E, Letter: L, Letter: L, Letter: O
Position 0: H, Position 1: E, Position 2: L...

Practice Question 3

String & Loop Challenge

Create a function that takes a string and returns it reversed using a for loop.

function reverseString(str) {
    let reversed = "";
    
    for (let i = str.length - 1; i >= 0; i--) {
        reversed += str[i];
    }
    
    return reversed;
}

console.log(reverseString("Hello")); // "olleH"

Practice Question 4

Do-While Challenge

Write a do-while loop that asks the user to guess a number (1-5) until they get it right. Use a predetermined correct answer.

const correctAnswer = 3;
let userGuess;

do {
    userGuess = parseInt(prompt("Guess a number between 1 and 5:"));
    
    if (userGuess !== correctAnswer) {
        alert("Wrong! Try again.");
    }
} while (userGuess !== correctAnswer);

alert("Correct! You guessed it!");

Advanced Example 🚀

Let's combine everything: loops, strings, and template literals!

const students = [
    { name: "Alice", grade: 95 },
    { name: "Bob", grade: 87 },
    { name: "Charlie", grade: 92 }
];

for (const student of students) {
    const letterGrade = student.grade >= 90 ? 'A' : 
                        student.grade >= 80 ? 'B' : 'C';
    
    console.log(`${student.name}: ${student.grade}% (${letterGrade})`);
}
Output:
Alice: 95% (A)
Bob: 87% (B)
Charlie: 92% (A)
